def with_output_buffer_with_haml_xss(*args, &block)
  res = with_output_buffer_without_haml_xss(*args, &block)
  case res
  when Array; res.map {|s| Haml::Util.html_safe(s)}
  when String; Haml::Util.html_safe(res)
  else; res
  end
end
